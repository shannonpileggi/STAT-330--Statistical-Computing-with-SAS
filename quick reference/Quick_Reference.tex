\Large{\ttb{FAQ's:}}
\normalsize{
\begin{enumerate}
\item How do I clear my log?
\bi
\item Make your log window active by clicking on it, and then click on the blank page symbol.
\ei
\item How do I clear my results window?
\bi
\item Submit \fbox{\ttt{ ODS HTML CLOSE; ODS HTML;}}
\item This stops SAS from writing more output to the current ``Results Viewer'', and then creates a new ``Results Viewer'' \ttb{when further output is created}.
\ei
\item How do I create that fancy shortcut key that submits my code, clears my log, and submits my results?
\bi
\item Go to Tools $\rightarrow$ Options $\rightarrow$ Keys (or \keys{F9})
\item In \keys{F12} type \ttb{\ttt{odsresults;clear;log;clear;wpgm;submit;log;top;}}
\ei
 \item  I closed my Explorer window.  How do I get it back?
  \bi
    \item Point-and-click to \ttt{View --> Contents Only}
    \item Point-and-click to \ttt{View --> Results}
 \ei
 \item How do I tile my SAS windows?
 \bi
    \item Point-and-click to \ttt{Window --> Tile Vertically/Horizontally}
 \ei
 \item What are the options I should use to help identify errors when writing macros?
 \item[]\fbox{\ttt{OPTIONS MPRINT MLOGIC SYMBOLGEN;}}
 }
\end{enumerate}


\Large{\ttb{Shortcut keys:}}
\normalsize{
\bi
\item To quickly comment/uncomment a section of code:
	\bi
	\item Select the code section by highlighting with the cursor and then use \keys{Ctrl} + \keys{/}
	\item To uncomment any selected code simply use \keys{Ctrl} + \keys{Shift} + \keys{/}
	\ei
\item To quickly change case in the SAS editor:
    \bi
    \item Convert selected text to upper case \keys{Ctrl} + \keys{Shift} + \keys{U}
    \item Convert selected text to lower case \keys{Ctrl} + \keys{Shift} + \keys{L}
    \ei
\ei
